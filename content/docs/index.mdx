---
title: Get started
description: How to set up Free Ship Fast with Better Auth and your choice of database.
---

<Steps>

<Step>
### 1. Clone the Repo

```sh
git clone https://github.com/MVP-LaunchPad-Labs/free-ship-fast-www.git [YOUR_APP_NAME]
cd [YOUR_APP_NAME]
```

<Callout type="info">
Choose your stack and checkout the corresponding branch in the next step.
</Callout>
</Step>

<Step>
### 2. Select Your Stack

<Tabs groupId="stack" items={["Better Auth + Postgres", "Better Auth + MongoDB", "Supabase"]} persist defaultIndex={0}>
  <Tab value="Better Auth + Postgres">
    <Callout>Use branch: <b>main (default)</b></Callout>
    ```sh
    bun install
    git remote remove origin
    ```
  </Tab>
  <Tab value="Better Auth + MongoDB">
    <Callout>Use branch: <b>better-auth-mongodb</b></Callout>
    ```sh
    git checkout better-auth-mongodb
    bun install
    git remote remove origin
    ```
  </Tab>
  <Tab value="Supabase">
    <Callout>Use branch: <b>supabase</b></Callout>
    ```sh
    git checkout supabase
    bun install
    git remote remove origin
    ```
  </Tab>
</Tabs>
</Step>

<Step>
### 3. Configure Environment Variables

Rename `.env.example` to `.env.local`:

```sh
mv .env.example .env.local
```

Set up your environment variables:

<Tabs groupId="env" items={["Postgres", "MongoDB", "Supabase"]} persist defaultIndex={0}>
  <Tab value="Postgres">
    ```txt title=".env"
    # Better Auth
    BETTER_AUTH_SECRET=your-random-secret-key
    BETTER_AUTH_URL=http://localhost:3000
    NEXT_PUBLIC_BETTER_AUTH_URL=http://localhost:3000
    
    # Google OAuth
    GOOGLE_CLIENT_ID=your-google-client-id
    GOOGLE_CLIENT_SECRET=your-google-client-secret
    
    # App URLs
    NEXT_PUBLIC_APP_URL=http://localhost:3000
    SITE_URL=http://localhost:3000
    
    # Database
    DATABASE_URL=postgresql://user:password@host:5432/database
    
    # Email (Resend)
    RESEND_API_KEY=your-resend-api-key
    RESEND_FROM_EMAIL=no-reply@yourdomain.com
    
    # Rate Limiting (Upstash Redis)
    UPSTASH_REDIS_REST_URL=your-upstash-redis-url
    UPSTASH_REDIS_REST_TOKEN=your-upstash-redis-token
    
    # Stripe
    STRIPE_SECRET_KEY=sk_test_xxx
    STRIPE_PUBLISHABLE_KEY=pk_test_xxx
    STRIPE_WEBHOOK_SECRET=whsec_xxx
    
    # Lemon Squeezy (optional)
    LEMON_SQUEEZY_SECRET_KEY=sk_live_xxx
    LEMON_SQUEEZY_STORE_ID=your-store-id
    LEMON_SQUEEZY_WEBHOOK_SECRET=whsec_xxx
    ```
  </Tab>
  <Tab value="MongoDB">
    ```txt title=".env"
    # Better Auth
    BETTER_AUTH_SECRET=your-random-secret-key
    BETTER_AUTH_URL=http://localhost:3000
    NEXT_PUBLIC_BETTER_AUTH_URL=http://localhost:3000
    
    # Google OAuth
    GOOGLE_CLIENT_ID=your-google-client-id
    GOOGLE_CLIENT_SECRET=your-google-client-secret
    
    # App URLs
    NEXT_PUBLIC_APP_URL=http://localhost:3000
    SITE_URL=http://localhost:3000
    
    # Database
    MONGODB_URI=mongodb+srv://user:password@cluster.mongodb.net/database
    
    # Email (Resend)
    RESEND_API_KEY=your-resend-api-key
    RESEND_FROM_EMAIL=no-reply@yourdomain.com
    
    # Rate Limiting (Upstash Redis)
    UPSTASH_REDIS_REST_URL=your-upstash-redis-url
    UPSTASH_REDIS_REST_TOKEN=your-upstash-redis-token
    
    # Stripe
    STRIPE_SECRET_KEY=sk_test_xxx
    STRIPE_PUBLISHABLE_KEY=pk_test_xxx
    STRIPE_WEBHOOK_SECRET=whsec_xxx
    
    # Lemon Squeezy (optional)
    LEMON_SQUEEZY_SECRET_KEY=sk_live_xxx
    LEMON_SQUEEZY_STORE_ID=your-store-id
    LEMON_SQUEEZY_WEBHOOK_SECRET=whsec_xxx
    ```
  </Tab>
  <Tab value="Supabase">
    ```txt title=".env"
    # Supabase
    NEXT_PUBLIC_SUPABASE_URL=https://your-project.supabase.co
    NEXT_PUBLIC_SUPABASE_ANON_KEY=your-anon-key
    SUPABASE_SERVICE_ROLE_KEY=your-service-role-key
    
    # App URLs
    NEXT_PUBLIC_APP_URL=http://localhost:3000
    SITE_URL=http://localhost:3000
    
    # Email (Resend)
    RESEND_API_KEY=your-resend-api-key
    RESEND_FROM_EMAIL=no-reply@yourdomain.com
    
    # Stripe
    STRIPE_SECRET_KEY=sk_test_xxx
    STRIPE_PUBLISHABLE_KEY=pk_test_xxx
    STRIPE_WEBHOOK_SECRET=whsec_xxx
    
    # Lemon Squeezy (optional)
    LEMON_SQUEEZY_SECRET_KEY=sk_live_xxx
    LEMON_SQUEEZY_STORE_ID=your-store-id
    LEMON_SQUEEZY_WEBHOOK_SECRET=whsec_xxx
    ```
  </Tab>
</Tabs>
</Step>

<Step>
### 4. Run the App

```sh
bun run dev
```

Open [http://localhost:3000](http://localhost:3000) in your browser.

<Callout type="info">
Requires Node 18.17 or higher (`node -v` to check)
</Callout>
</Step>
